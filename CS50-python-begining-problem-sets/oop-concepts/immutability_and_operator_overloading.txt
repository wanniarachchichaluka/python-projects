Instance Attributes, Class Attributes, Instance Methods, Class Methods
The Core Difference
Instance = Belongs to each specific object (unique)

Class = Belongs to the class itself (shared by ALL objects)

Real-World Analogy: School
Think of a School class:

Class attribute: School name, address (same for ALL students)

Instance attribute: Each student's name, grade, ID (unique per student)

Let me show you with a simple example:

python
class Student:
    # CLASS ATTRIBUTES - shared by ALL students
    school_name = "Springfield Elementary"
    total_students = 0
    
    def __init__(self, name, grade):
        # INSTANCE ATTRIBUTES - unique to each student
        self.name = name
        self.grade = grade
        self.id = Student.total_students + 1  # Using class attribute
        
        # Update class attribute
        Student.total_students += 1
    
    # INSTANCE METHOD - works with instance data
    def get_info(self):
        return f"{self.name} is in grade {self.grade}"
    
    # INSTANCE METHOD - can access both instance and class attributes
    def introduce(self):
        return f"I'm {self.name} from {Student.school_name}"
    
    # CLASS METHOD - works with class data, NOT instance data
    @classmethod
    def get_school_info(cls):
        return f"School: {cls.school_name}, Total Students: {cls.total_students}"
    
    # CLASS METHOD - alternative constructor
    @classmethod
    def create_default_student(cls):
        return cls("Default Student", "Unknown")

# Let's create instances (objects)
student1 = Student("Alice", 10)
student2 = Student("Bob", 11)
student3 = Student("Charlie", 10)

# INSTANCE ATTRIBUTES - each student has unique values
print("=== INSTANCE ATTRIBUTES (Unique per object) ===")
print(f"Student1: {student1.name}, Grade: {student1.grade}, ID: {student1.id}")
print(f"Student2: {student2.name}, Grade: {student2.grade}, ID: {student2.id}")
print(f"Student3: {student3.name}, Grade: {student3.grade}, ID: {student3.id}")

# CLASS ATTRIBUTES - shared, same value for all objects
print("\n=== CLASS ATTRIBUTES (Shared by all) ===")
print(f"Student1 sees: {student1.school_name}")
print(f"Student2 sees: {student2.school_name}")
print(f"Student3 sees: {student3.school_name}")
print(f"Access via class: {Student.school_name}")

# INSTANCE METHODS - called on specific objects
print("\n=== INSTANCE METHODS (Called on objects) ===")
print(student1.get_info())
print(student2.get_info())
print(student1.introduce())

# CLASS METHODS - called on the class itself, not on objects
print("\n=== CLASS METHODS (Called on class) ===")
print(Student.get_school_info())

# Alternative constructor using class method
print("\n=== CLASS METHOD AS ALTERNATIVE CONSTRUCTOR ===")
default_student = Student.create_default_student()
print(default_student.get_info())

class Example:
    # CLASS ATTRIBUTE (shared)
    class_var = "I belong to the class"
    
    def __init__(self, value):
        # INSTANCE ATTRIBUTE (unique)
        self.instance_var = value
    
    # INSTANCE METHOD (can access both)
    def instance_method(self):
        return f"Instance: {self.instance_var}, Class: {self.class_var}"
    
    # CLASS METHOD (only accesses class stuff)
    @classmethod
    def class_method(cls):
        return f"Class only: {cls.class_var}"

obj2 = Example("B")

print(f"obj1: {obj1.instance_var}")  # A
print(f"obj2: {obj2.instance_var}")  # B

print("\n=== CLASS ATTRIBUTES are the SAME ===")
print(f"obj1: {obj1.class_var}")  # I belong to the class
print(f"obj2: {obj2.class_var}")  # I belong to the class
print(f"Class: {Example.class_var}")  # I belong to the class

print("\n=== INSTANCE METHODS need an object ===")
print(obj1.instance_method())  # Instance: A, Class: I belong to the class

print("\n=== CLASS METHODS work on class directly ===")
print(Example.class_method())  # Class only: I belong to the class
When to Use What:
Type	When to Use	Can Access
Instance Attribute	Data unique to each object (name, age, ID)	Only that instance
Class Attribute	Data shared by all objects (company name, tax rate)	All instances + class
Instance Method	Actions that need specific object data	Instance + Class attributes

class BankAccount:
    # CLASS ATTRIBUTES
    bank_name = "Python Bank"
    interest_rate = 0.02  # 2% interest for ALL accounts
    total_accounts = 0
    
    def __init__(self, owner, initial_balance):
        # INSTANCE ATTRIBUTES
        self.owner = owner
        self.balance = initial_balance
        self.account_number = BankAccount.total_accounts + 1
        
        BankAccount.total_accounts += 1
    
    # INSTANCE METHOD
    def deposit(self, amount):
        self.balance += amount
        return f"{self.owner}'s new balance: ${self.balance}"
    
    # INSTANCE METHOD
    def withdraw(self, amount):
        if amount <= self.balance:
            self.balance -= amount
            return f"{self.owner} withdrew ${amount}"
        return "Insufficient funds!"
    
    # INSTANCE METHOD
    def get_info(self):
        return f"Account {self.account_number}: {self.owner} - ${self.balance}"
    
    # CLASS METHOD - to get bank-wide info
    @classmethod
    def get_bank_info(cls):
        return f"{cls.bank_name}: {cls.total_accounts} accounts, {cls.interest_rate*100}% interest"
    
    # CLASS METHOD - alternative constructor
    @classmethod
    def create_joint_account(cls, owner1, owner2, initial_balance):
        joint_owner = f"{owner1} & {owner2}"
        return cls(joint_owner, initial_balance)

# Create accounts
acc1 = BankAccount("Alice", 1000)
acc2 = BankAccount("Bob", 500)
acc3 = BankAccount.create_joint_account("Alice", "Bob", 2000)

# Instance methods (unique per account)
print(acc1.deposit(200))      # Alice's new balance: $1200
print(acc2.withdraw(100))     # Bob withdrew $100
print(acc3.get_info())         # Account 3: Alice & Bob - $2000

# Class method (bank-wide info)
print(BankAccount.get_bank_info())  # Python Bank: 3 accounts, 2.0% interest

# Class attributes are shared
print(f"All accounts have same bank: {acc1.bank_name} = {acc2.bank_name}")
Important Notes:
Instance methods need self parameter

Class methods need cls parameter and @classmethod decorator

Instance methods can access class attributes but not vice versa

Class methods cannot access instance attributes (they don't know which object)

1. Immutability
What is Immutability?
An immutable object cannot be changed after it's created. Once created, its state is fixed forever. If you want to "change" it, you actually create a NEW object.

Real-World Analogy
Think of a birth date - once a person is born, their birth date never changes. You can't "modify" it; you can only create a new record with a different date.

Immutable vs Mutable in Python
python
# IMMUTABLE types (CANNOT be changed)
# - int, float, bool
# - str (string)
# - tuple
# - frozenset

# MUTABLE types (CAN be changed)
# - list
# - dict
# - set
# - custom objects (by default)

print("=== STRINGS (IMMUTABLE) ===")
text = "Hello"
print(f"Original text: {text}")
print(f"Original memory ID: {id(text)}")

# This creates a NEW string, doesn't modify the original
text = text + " World"
print(f"Modified text: {text}")
print(f"New memory ID: {id(text)}")  # Different ID! New object created

# Trying to modify a string directly gives error
try:
    text[0] = "J"  # This will fail
except TypeError as e:
    print(f"Error: {e}")

print("\n=== LISTS (MUTABLE) ===")
my_list = [1, 2, 3]
print(f"Original list: {my_list}")
print(f"Original memory ID: {id(my_list)}")

# This modifies the SAME list
my_list.append(4)
print(f"Modified list: {my_list}")
print(f"Memory ID after append: {id(my_list)}")  # Same ID! Same object

my_list[0] = 99  # This works for mutable objects
print(f"After changing element: {my_list}")

class ImmutablePoint:
    """An immutable point class - once created, cannot change"""
    
    def __init__(self, x, y):
        # Private attributes (convention to indicate don't modify)
        self._x = x
        self._y = y
    
    # Properties provide read-only access
    @property
    def x(self):
        return self._x
    
    @property
    def y(self):
        return self._y
    
    # To "change" a point, we create a new point
    def move(self, dx, dy):
        """Returns a NEW point at new position (doesn't modify original)"""
        return ImmutablePoint(self._x + dx, self._y + dy)
    
    def __str__(self):
        return f"Point({self._x}, {self._y})"

# Using immutable class
p1 = ImmutablePoint(3, 4)
print(f"p1: {p1}")
print(f"p1.x = {p1.x}")

# Cannot modify - this will fail
try:
    p1._x = 10  # Technically possible but breaking convention
    # Or trying to set via property
    # p1.x = 10  # AttributeError: can't set attribute
except AttributeError as e:
    print(f"Can't modify: {e}")

# To change, create new object
p2 = p1.move(2, 3)
print(f"p1 after move (unchanged): {p1}")
print(f"New point p2: {p2}")
print(f"Are they the same object? {p1 is p2}")  # False - different objects

2. Operator Overloading
What is Operator Overloading?
It allows you to define how operators (+, -, *, ==, <, etc.) work with your custom classes. You can make your objects behave like built-in types.

Real-World Analogy
The + operator means "add" for numbers, but "concatenate" for strings. Operator overloading lets you define what + means for your custom class (e.g., adding two Vector objects).

Common Magic Methods (Operators)
python
class Vector:
    """A 2D vector with operator overloading"""
    
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    # String representation
    def __str__(self):
        return f"Vector({self.x}, {self.y})"
    
    # repr() representation
    def __repr__(self):
        return f"Vector({self.x}, {self.y})"
    
    # OPERATOR OVERLOADING METHODS
    
    # Addition: v1 + v2
    def __add__(self, other):
        return Vector(self.x + other.x, self.y + other.y)
    
    # Subtraction: v1 - v2
    def __sub__(self, other):
        return Vector(self.x - other.x, self.y - other.y)
    
    # Multiplication (scalar): v * n
    def __mul__(self, scalar):
        return Vector(self.x * scalar, self.y * scalar)
    
    # Right multiplication: n * v
    def __rmul__(self, scalar):
        return self.__mul__(scalar)
    
    # Equality: v1 == v2
    def __eq__(self, other):
        return self.x == other.x and self.y == other.y
    
    # Less than: v1 < v2 (based on magnitude)
    def __lt__(self, other):
        return (self.x**2 + self.y**2) < (other.x**2 + other.y**2)
    
    # Length: len(v)
    def __len__(self):
        return int((self.x**2 + self.y**2) ** 0.5)
    
    # Indexing: v[0], v[1]
    def __getitem__(self, index):
        if index == 0:
            return self.x
        elif index == 1:
            return self.y
        else:
            raise IndexError("Vector index out of range")
    
    # Negation: -v
    def __neg__(self):
        return Vector(-self.x, -self.y)

# Demonstrating operator overloading
v1 = Vector(3, 4)
v2 = Vector(1, 2)

print("=== VECTOR OPERATIONS ===")
print(f"v1: {v1}")
print(f"v2: {v2}")
print(f"v1 + v2 = {v1 + v2}")      # Vector(4, 6)
print(f"v1 - v2 = {v1 - v2}")      # Vector(2, 2)
print(f"v1 * 3 = {v1 * 3}")        # Vector(9, 12)
print(f"2 * v1 = {2 * v1}")        # Vector(6, 8)
print(f"v1 == v2? {v1 == v2}")     # False
print(f"v1 < v2? {v1 < v2}")       # False (v1 magnitude 5, v2 magnitude ~2.2)
print(f"Length of v1: {len(v1)}")   # 5 (sqrt(9+16))
print(f"v1[0] = {v1[0]}, v1[1] = {v1[1]}")  # 3, 4
print(f"-v1 = {-v1}")              # Vector(-3, -4)

class Money:
    """Money class with operator overloading"""
    
    def __init__(self, amount, currency="USD"):
        self.amount = amount
        self.currency = currency
    
    def __str__(self):
        return f"{self.amount} {self.currency}"
    
    # Adding money: m1 + m2
    def __add__(self, other):
        if self.currency != other.currency:
            raise ValueError(f"Cannot add {self.currency} and {other.currency}")
        return Money(self.amount + other.amount, self.currency)
    
    # Subtraction: m1 - m2
    def __sub__(self, other):
        if self.currency != other.currency:
            raise ValueError(f"Cannot subtract {self.currency} and {other.currency}")
        return Money(self.amount - other.amount, self.currency)
    
    # Equality: m1 == m2
    def __eq__(self, other):
        return self.amount == other.amount and self.currency == other.currency
    
    # Greater than: m1 > m2
    def __gt__(self, other):
        if self.currency != other.currency:
            raise ValueError(f"Cannot compare {self.currency} and {other.currency}")
        return self.amount > other.amount
    
    # Less than: m1 < m2
    def __lt__(self, other):
        if self.currency != other.currency:
            raise ValueError(f"Cannot compare {self.currency} and {other.currency}")
        return self.amount < other.amount
    
    # Absolute value: abs(m)
    def __abs__(self):
        return Money(abs(self.amount), self.currency)
    
    # Negation: -m
    def __neg__(self):
        return Money(-self.amount, self.currency)

# Demonstration
salary = Money(5000, "USD")
bonus = Money(1000, "USD")
tax = Money(1200, "USD")

print("=== MONEY OPERATIONS ===")
print(f"Salary: {salary}")
print(f"Bonus: {bonus}")
print(f"Salary + Bonus = {salary + bonus}")  # 6000 USD
print(f"Salary - Tax = {salary - tax}")       # 3800 USD
print(f"Salary > Bonus? {salary > bonus}")    # True
print(f"Salary == 5000 USD? {salary == Money(5000, 'USD')}")  # True
print(f"Negative money: {-salary}")           # -5000 USD
print(f"Absolute value: {abs(Money(-300, 'USD'))}")  # 300 USD
